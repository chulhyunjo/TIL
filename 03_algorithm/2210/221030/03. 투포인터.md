# 03. 투포인터

```python
# 연속된 자연수 합 구하기 boj2018
n = int(input())
result = 1

start_index = 1
end_index = 1
sum1 = 1
while end_index < n:
    if sum1 > n:
        sum1 -= start_index
        start_index += 1
    elif sum1 < n:
        end_index += 1
        sum1 += end_index
    else:
        result += 1
        end_index += 1
        sum1 += end_index
print(result)
```



```python
# 주몽 boj1940
n = int(input())
m = int(input())
arr = sorted(list(map(int,input().split())))

start_index = 0
end_index = n-1
result = 0

sum1 = arr[start_index] + arr[end_index]
while start_index < end_index:
    if sum1 > m:
        sum1 -= arr[end_index]
        end_index -= 1
        sum1 += arr[end_index]
    elif sum1 < m:
        sum1 -= arr[start_index]
        start_index += 1
        sum1 += arr[start_index]
    else:
        result += 1
        sum1 -= arr[start_index]
        start_index += 1
        sum1 += arr[start_index]
print(result)
```
