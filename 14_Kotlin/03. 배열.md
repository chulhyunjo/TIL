```kotlin
fun main(args: Array<String>) {
    val squares = arrayOf(1, 4, 9, 16)
    squares.size // 4
    squares.lastIndex // 3
    squares[3] // 16
    squares[1] // 4

    val numbers = squares.copyOf() // 배열 복사
    squares.copyOf(2) // 뒤가 짤림 1, 4
    squares.copyOf(5) // 부족한 부분은 0으로 채우짐

    val b = intArrayOf(1, 2, 3) + 4 // 원소 추가
    val c = intArrayOf(1, 2, 3) + intArrayOf(5, 6) // 다른 배열 추가
    
    intArrayOf(1, 2, 3) == intArrayOf(1, 2, 3) // false
    intArrayOf(1, 2, 3).contentEquals(intArrayOf(1, 2, 3)) // true
    
    intArrayOf(1, 2, 3).indexOf(4) // -1반환
    intArrayOf(1, 2, 3).contentToString() // 리스트 출력
}
```
