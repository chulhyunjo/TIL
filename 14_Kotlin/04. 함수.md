import kotlin.math.PI

fun circleArea(radius: Double):Double {
    return PI * radius * radius
}
fun main(args: Array<String>) {
    print("Enter radius: ")
    val radius = readLine()!!.toDouble()
    println("넓이 : ${circleArea(radius)}")
}

```kotlin
fun swap(s:String, from:Int, to:Int) :String {
    val chars = s.toCharArray() // 배열로 변환
    
    val tmp = chars[from]
    chars[from] = chars[to]
    chars[to] = tmp
    return chars.concatToString() // 합치기
}

fun main(){
    println(swap("Hello", 1, 2)) // Hlelo
    println(swap("Hello", to=2, from=1)) // Hlelo
}

```

- 디폴트 값 지정 가능

```kotlin
import kotlin.math.max
import kotlin.math.min

fun restrictToRange(from:Int = Int.MIN_VALUE, to: Int = Int.MAX_VALUE, what:Int) : Int{
    return max(from, min(to, what))
}

fun main(){
    println(restrictToRange(10, what= 1))
}
```



- 인자의 개수가 정해져있지 않으면 vararg를 앞에 붙인
- 스프레드는 배열을 복사하는 것이다.

```kotlin
fun printSorted(vararg items: Int){
    items.sort()
    println(items.contentToString())// [1,2,6,10]
}

fun main(){
    val items: IntArray = intArrayOf(6, 2, 10, 1)
    printSorted(*items)
    println(items.contentToString()) // [6, 2, 10, 1]
}
```



```kotlin
fun change(vararg items: IntArray){
    items[0][0] = 100
    println(items.contentToString())
}

fun main(){
    val a: IntArray = intArrayOf(1, 2, 3)
    val b: IntArray = intArrayOf(4, 5, 6)
    change(a, b)
    println(a.contentToString()) // 100,2,3
    println(b.contentToString()) // 4, 5, 6
}
```
